## With validation
<div>
  <FormGroup
    label={
      <div>Email Address <span className="text-primary-500">*</span></div>
    }
    helperText="This email address is used for send important emails">
    <InputGroup
      type="email"
      isValid={true}
      placeholder="Enter your email" />
      <FormGroup.Message
        type="success"
        className="mt-1">Congratulations!!</FormGroup.Message>
  </FormGroup>
  <FormGroup
    className="pt-4"
    label={
      <div>Password <span className="text-primary-500">*</span></div>
    }>
    <InputGroup
      isValid={false}
      type="password" />
    <FormGroup.Message
      type="error"
      className="mt-1">Please verify your credentials</FormGroup.Message>
  </FormGroup>
</div>
